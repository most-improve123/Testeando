  <h1>Verificador de Certificados</h1>
  <form id="formularioVerificar">
    <input type="text" id="idCertificado" placeholder="Ingresa el ID o Hash del certificado" required />
    <button type="submit">Verificar</button>
  </form>
  <div id="resultado"></div>
  <script>
  document.getElementById('formularioVerificar').addEventListener('submit', function(e) {
    e.preventDefault();
    const entrada = document.getElementById('idCertificado').value.trim();
    const certificados = JSON.parse(localStorage.getItem('certificados') || '[]');
    
    // Buscar por ID o por hash
    const encontrado = certificados.find(cert => cert.id === entrada || cert.hash === entrada);
    const resultado = document.getElementById('resultado');
    resultado.innerHTML = '';

    if (encontrado) {
      resultado.innerHTML = `
        <h2>✅ Certificado Válido</h2>
        <p><strong>Nombre:</strong> ${encontrado.nombre}</p>
        <p><strong>Curso:</strong> ${encontrado.curso}</p>
        <p><strong>Fecha:</strong> ${encontrado.fecha}</p>
        <p><strong>ID:</strong> ${encontrado.id}</p>
        <p><strong>Hash:</strong> ${encontrado.hash}</p>
      `;
    } else {
      resultado.innerHTML = '<h2>❌ Certificado no encontrado o inválido</h2>';
    }
  });
</script>
